---
title: "Get an agent"
api: "PUT https://api.mintlify.com/api/get-agent"
description: "This endpoint returns the information of an existing agent."
---

### Body

<ParamField body="agent_id" type="string">
  This is the ID of the agent.
</ParamField>

### Response

<ResponseField name="success" type="number">
  Indicates whether the call was successful. 1 if successful, 0 if not.
</ResponseField>

<ResponseField name="agent" type="object">
  This is an object of the agent.

  <Expandable title="Fields">
  
    <ResponseField name="name" type="string">
      This is the name of an agent.
    </ResponseField>

    <ResponseField name="age" type="number">
      This is the age of an agent.
    </ResponseField>

    <ResponseField name="personality" type="string">
      This is the personality of an agent.
    </ResponseField>

    <ResponseField name="initial_status" type="string">
      This is the initial status of an agent.
    </ResponseField>

    <ResponseField name="initial_memory" type="object">
      This is the initial memory of an agent.
    </ResponseField>

    <ResponseField name="avatar" type="object" optional>
      This is the avatar object of an agent
      <Expandable title="Fields">
        <ResponseField name="cloud_path" type="string">
          This is the path to the avatar in cloud.
        </ResponseField>

        <ResponseField name="local_path" type="string">
          This is the path to the avatar in local.
        </ResponseField>
      <Expandable title="Fields">
    </ResponseField>

  </Expandable>

</ResponseField>

<RequestExample>

```bash Example Request
curl --location --request PUT 'https://api.mintlify.com/api/get-agent' \
--header 'Content-Type: application/json' \
--header 'Authorization: Token <token>' \
--data-raw '{
    "agent_id": "xxxxxxxxxxxxxx"
}'
```

</RequestExample>
