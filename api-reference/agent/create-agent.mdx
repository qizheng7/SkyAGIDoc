---
title: "Create an agent"
api: "PUT https://api.mintlify.com/api/create-agent"
description: "This endpoint creates an agent."
---

### Body

<ParamField body="user_id" type="str">
  This is the ID of the user.
</ParamField>

<ParamField body="agent" type="object">
  This is an object of the agent involved in a chat
  session. The agent contains the meta information and the initial
  memory associated with it.

  <Expandable title="Fields">
  
    <ResponseField name="name" type="string">
      This is the name of an agent.
    </ResponseField>

    <ResponseField name="age" type="number">
      This is the age of an agent.
    </ResponseField>

    <ResponseField name="personality" type="string">
      This is the personality of an agent.
    </ResponseField>

    <ResponseField name="status" type="string">
      This is the initial status of an agent.
    </ResponseField>

    <ResponseField name="memory" type="object">
      This is the initial memory of an agent.
    </ResponseField>

    <ResponseField name="avatar" type="object" optional>
      This is the avatar object of an agent
      <Expandable title="Fields">
        <ResponseField name="cloud_path" type="string">
          This is the path to the avatar in cloud.
        </ResponseField>

        <ResponseField name="local_path" type="string">
          This is the path to the avatar in local.
        </ResponseField>
      <Expandable title="Fields">
    </ResponseField>

  </Expandable>

</ParamField>

### Response

<ResponseField name="success" type="number">
  Indicates whether the call was successful. 1 if successful, 0 if not.
</ResponseField>

<ResponseField name="agent_id" type="string">
  The ID of the created agent.
</ResponseField>
