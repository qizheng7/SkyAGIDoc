---
title: "Get all agents of a user ID"
api: "PUT https://api.mintlify.com/api/get-agents"
description: "This endpoint returns the full lists of agents associated with a user ID."
---

### Body

<ParamField body="user_id" type="string">
  This is the ID of the user.
</ParamField>

### Response

<ResponseField name="success" type="number">
  Indicates whether the call was successful. 1 if successful, 0 if not.
</ResponseField>

<ResponseField name="agents" type="object">
  This is an array of the agent objects.

  <Expandable title="Fields">

    <ResponseField name="id" type="string">
      This is the ID of an agent.
    </ResponseField>


    <ResponseField name="name" type="string">
      This is the name of an agent.
    </ResponseField>

    <ResponseField name="age" type="number">
      This is the age of an agent.
    </ResponseField>

    <ResponseField name="personality" type="string">
      This is the personality of an agent.
    </ResponseField>

    <ResponseField name="initial_status" type="string">
      This is the initial status of an agent.
    </ResponseField>

    <ResponseField name="initial_memory" type="object">
      This is the initial memory of an agent.
    </ResponseField>

  </Expandable>

</ResponseField>

<RequestExample>

```bash Example Request
curl --location --request PUT 'https://api.mintlify.com/api/get-agents' \
--header 'Content-Type: application/json' \
--header 'Authorization: Token <token>' \
--data-raw '{
    "user_id": "xxxxxxxxxxxxxx"
}'
```

</RequestExample>
