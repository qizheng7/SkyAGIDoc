---
title: "Send system message"
api: "PUT https://api.mintlify.com/api/send-system-message"
description: "This endpoint sends a system message to an agent."
---

### Body

<ParamField body="conversation_id" type="string">
  This is the ID of the conversation.
</ParamField>

<ParamField body="initiate_agent" type="string">
  This is the ID of the agent who will initiate the conversation.
</ParamField>

<ParamField body="initiate_agent_model" type="object">
  This is the model that the initiate agent uses, along with the model key.
  <Expandable title="Fields">
    <ResponseField name="name" type="string">
      This is the selected model name.
    </ResponseField>

    <ResponseField name="token" type="string">
      This is the model token or API key.
    </ResponseField>

  </Expandable>

</ParamField>

<ParamField body="recipient_agent" type="string">
  This is the ID of the agent who receives the conversation.
</ParamField>

<ParamField body="recipient_agent_model" type="object">
  This is the model that the recipient agent uses, along with the model key.
  <Expandable title="Fields">
    <ResponseField name="name" type="string">
      This is the selected model name.
    </ResponseField>

    <ResponseField name="token" type="string">
      This is the model token or API key.
    </ResponseField>

  </Expandable>

</ParamField>

<ParamField body="message" type="string" optional>
  This is the system message sent to the initiate agent about whether it wants
  to initiate a message to the recipient agent.
</ParamField>

### Response

<ResponseField name="success" type="number">
  Indicates whether the call was successful. 1 if successful, 0 if not.
</ResponseField>

<ResponseField name="resp_msg" type="string">
  This is the response from the initiate agent to either the system or the
  recipient agent.
</ResponseField>
