---
title: "Create conversation"
api: "PUT https://api.mintlify.com/api/create-conversation"
description: "This endpoint creates a new conversation session"
---

### Body

<ParamField body="name" type="string">
  This is the name of the new conversation session.
</ParamField>

<ParamField body="user_id" type="string">
  This is the ID of current user.
</ParamField>

<ParamField body="agents" type="object">
  This is an array of the agents involved in the conversation.
  <Expandable title="Fields">
    <ResponseField name="id" type="string">
      This is the ID of an agent.
    </ResponseField>

    <ParamField body="embedding_model_settings" type="object">
        This is the embedding model settings.
        <Expandable title="Fields">
            <ResponseField name="type" type="string">
                This is the type of the embedding model.
            </ResponseField>

            <ResponseField name="provider" type="string">
                This is the provider of the embedding model.
            </ResponseField>

            <ResponseField name="name" type="string">
                This is the name of the embedding model.
            </ResponseField>

            <ResponseField name="args" type="object">
                This is the arguments of the embedding model, required by Langchain. The arguments can be varied.
            </ResponseField>

            <ResponseField name="embeddingSize" type="number">
                This is the embedding size of the embedding model.
            </ResponseField>
        </Expandable>
    </ParamField>

  </Expandable>
</ParamField>

<ParamField body="user_agent_ids" type="object">
  This is an array of the agents played by the user.
</ParamField>

### Response

<ResponseField name="success" type="number">
  Indicates whether the call was successful. 1 if successful, 0 if not.
</ResponseField>

<ResponseField name="conversation_id" type="string">
  The ID of the created conversation.
</ResponseField>
