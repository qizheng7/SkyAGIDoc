---
title: "Get a conversation session"
api: "PUT https://api.mintlify.com/api/get-conversation"
description: "This endpoint returns a conversation session."
---

### Body

<ParamField body="conversation_id" type="string">
  This is the ID of the conversation.
</ParamField>

### Response

<ResponseField name="success" type="number">
  Indicates whether the call was successful. 1 if successful, 0 if not.
</ResponseField>

<ResponseField name="name" type="string">
  This is the name of the conversation.
</ResponseField>

<ResponseField name="agent_ids" type="object">
  This is the array of agent ids involved.
</ResponseField>

<ResponseField name="user_agent_ids" type="object">
  This is the array of user agent ids involved.
</ResponseField>

<ResponseField name="messages" type="object">
  This is the array of the messages. Each message object is as below.

  <Expandable title="Fields">
  
    <ResponseField name="initiate_agent_id" type="string">
      This is the ID of the initiate agent.
    </ResponseField>

    <ResponseField name="recipient_agent_id" type="string">
      This is the ID of the recipient agent.
    </ResponseField>

    <ResponseField name="create_time" type="timestamp">
      This is the time when the message was created.
    </ResponseField>

    <ResponseField name="content" type="string">
      This is the message content.
    </ResponseField>

  </Expandable>

</ResponseField>
