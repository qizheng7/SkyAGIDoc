---
title: "Create direct chat"
api: "POST https://api.mintlify.com/api/chat"
description: "This endpoint creates a direct chat with an agent"
---

### Body

<ParamField body="agent" type="object">
  This is an object of the agent involved in a chat
  session. The agent contains the meta information and the memory
  associated with it.

  <Expandable title="Fields">
  
    <ResponseField name="agent_name" type="string">
      This is the name of an agent.
    </ResponseField>

    <ResponseField name="age" type="number">
      This is the age of an agent.
    </ResponseField>

    <ResponseField name="personality" type="string">
      This is the personality of an agent.
    </ResponseField>

    <ResponseField name="status" type="string">
      This is the current status of an agent.
    </ResponseField>

    <ResponseField name="memory" type="object">
      This is the memory of an agent.
    </ResponseField>

  </Expandable>

</ParamField>

<ParamField body="model" type="object">
  This is a JSON object of the user selected large language model. It contains
  the model name and the api key or token.
  
  // there is a bug that expandable fields need to be wrapped as ResponseField
  <Expandable title="Fields">
    <ResponseField name="name" type="string">
      This is the selected model name.
    </ResponseField>

    <ResponseField name="token" type="string">
      This is the model token or API key.
    </ResponseField>

  </Expandable>
</ParamField>

### Response

<ResponseField name="success" type="number">
  Indicates whether the call was successful. 1 if successful, 0 if not.
</ResponseField>

<ResponseField name="chat_id" type="string">
  The ID of the created chat.
</ResponseField>

<RequestExample>

```bash Example Request
--header 'Content-Type: application/json' \
--header 'Authorization: Token <token>' \
--data-raw '{
    "agents": [
      {
        "name": "Penny",
        "age": 22,
        "personality": "Outgoing, friendly, compassionate, assertive, and determined",
        "status": "Penny is at the Cheesecake Factory",

          "Penny is originally from a small town in Nebraska and moved....",
          "Penny works as a waitress at the Cheesecake Factory, a job...",

        ]
      }
    ],

      "name": "gpt-3.5-turbo",
      "token": "xxxxxxxxxxxx",
    }
}'
```

</RequestExample>

</ResponseExample>
